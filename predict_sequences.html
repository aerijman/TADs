def fasta_header_extract(fileName):
    headers, sequences = [], []
    for i in open(fileName):
        if i[0]=='>':
            tmp = i.strip().split('_')
            headers.append(tmp)
            continue
        else:
            sequences.append(i.strip())
    values = np.array(headers)[:,1:6].astype(int)
    return pd.DataFrame(values, index=sequences, columns = ['bg','bin1','bin2','bin3','bin4'])

# function to compute kappa
import numpy as np


def neg(seq): return (seq.count('E')+seq.count('D'))/len(seq)
def pos(seq): return (seq.count('R')+seq.count('K')+0.5*seq.count('H')) / len(seq)


def sigma(seq):
    p = pos(seq)
    n = neg(seq)
    if p==0 and n==0: return 0
    return ((p-n)**2 / (p+n))

def Lambda(seq, Nblob):   #Nblob=5 if there are no prolines
    blobs = [seq[i:i+Nblob] for i in range(len(seq)-Nblob)]
    s = sigma(seq) 
    return np.sum( [(sigma(i)-s)**2 for i in blobs] ) / Nblob

def polarize(seq):
    posit = [i for i in seq if i=='R' or i=='K']
    negat = [i for i in seq if i=='D' or i=='E']
    H = [i for i in seq if i=='H']
    neutral = [i for i in seq if i not in posit+negat+H]
    return ''.join(posit+H+neutral+negat)

def kappa(seq):
    pol_seq = polarize(seq)
    k5 = Lambda(seq,5) / Lambda(pol_seq,5)
    k6 = Lambda(seq,6) / Lambda(pol_seq,6)
    return np.mean([k5,k6])
